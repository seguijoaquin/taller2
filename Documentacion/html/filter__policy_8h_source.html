9 // A database can be configured with a custom FilterPolicy object.
10 // This object is responsible for creating a small filter from a set
11 // of keys. These filters are stored in rocksdb and are consulted
12 // automatically by rocksdb to decide whether or not to read some
13 // information from disk. In many cases, a filter can cut down the
14 // number of disk seeks form a handful to a single disk seek per
15 // DB::Get() call.
16 //
17 // Most people will want to use the builtin bloom filter support (see
18 // NewBloomFilterPolicy() below).

56 // We add a new format of filter block called full filter block
57 // This new interface gives you more space of customization
58 //
59 // For the full filter block, you can plug in your version by implement
60 // the FilterBitsBuilder and FilterBitsReader
61 //
62 // There are two sets of interface in FilterPolicy
63 // Set 1: CreateFilter, KeyMayMatch: used for blockbased filter
64 // Set 2: GetFilterBitsBuilder, GetFilterBitsReader, they are used for
65 // full filter.
66 // Set 1 MUST be implemented correctly, Set 2 is optional
67 // RocksDB would first try using functions in Set 2. if they return nullptr,
68 // it would use Set 1 instead.
69 // You can choose filter type in NewBloomFilterPolicy
70 class FilterPolicy {
71  public:
72  virtual ~FilterPolicy();
73 
74  // Return the name of this policy. Note that if the filter encoding
75  // changes in an incompatible way, the name returned by this method
76  // must be changed. Otherwise, old incompatible filters may be
77  // passed to methods of this type.
78  virtual const char* Name() const = 0;
79 
80  // keys[0,n-1] contains a list of keys (potentially with duplicates)
81  // that are ordered according to the user supplied comparator.
82  // Append a filter that summarizes keys[0,n-1] to *dst.
83  //
84  // Warning: do not change the initial contents of *dst. Instead,
85  // append the newly constructed filter to *dst.
86  virtual void CreateFilter(const Slice* keys, int n, std::string* dst)
87  const = 0;
88 
89  // "filter" contains the data appended by a preceding call to
90  // CreateFilter() on this class. This method must return true if
91  // the key was in the list of keys passed to CreateFilter().
92  // This method may return true or false if the key was not on the
93  // list, but it should aim to return false with a high probability.
94  virtual bool KeyMayMatch(const Slice& key, const Slice& filter) const = 0;
95 
96  // Get the FilterBitsBuilder, which is ONLY used for full filter block
97  // It contains interface to take individual key, then generate filter
98  virtual FilterBitsBuilder* GetFilterBitsBuilder() const {
99  return nullptr;
100  }
101 
102  // Get the FilterBitsReader, which is ONLY used for full filter block
103  // It contains interface to tell if key can be in filter
104  // The input slice should NOT be deleted by FilterPolicy
105  virtual FilterBitsReader* GetFilterBitsReader(const Slice& contents) const {
106  return nullptr;
107  }
108 };

110 // Return a new filter policy that uses a bloom filter with approximately
111 // the specified number of bits per key.
112 //
113 // bits_per_key: bits per key in bloom filter. A good value for bits_per_key
114 // is 10, which yields a filter with ~ 1% false positive rate.
115 // use_block_based_builder: use block based filter rather than full fiter.
116 // If you want to builder full filter, it needs to be set to false.
117 //
118 // Callers must delete the result after any database that is using the
119 // result has been closed.
120 //
121 // Note: if you are using a custom comparator that ignores some parts
122 // of the keys being compared, you must not use NewBloomFilterPolicy()
123 // and must provide your own FilterPolicy that also ignores the
124 // corresponding parts of the keys. For example, if the comparator
125 // ignores trailing spaces, it would be incorrect to use a
126 // FilterPolicy (like NewBloomFilterPolicy) that does not ignore
127 // trailing spaces in keys.
128 extern const FilterPolicy* NewBloomFilterPolicy(int bits_per_key,
129  bool use_block_based_builder = true);
130 }
