19 // Abstract base class that defines the basic interface for a write batch.
20 // See WriteBatch for a basic implementation and WrithBatchWithIndex for an
21 // indexed implemenation.
22 class WriteBatchBase {
23  public:
24  virtual ~WriteBatchBase() {}
25 
26  // Store the mapping "key->value" in the database.
27  virtual void Put(ColumnFamilyHandle* column_family, const Slice& key,
28  const Slice& value) = 0;
29  virtual void Put(const Slice& key, const Slice& value) = 0;
30 
31  // Variant of Put() that gathers output like writev(2). The key and value
32  // that will be written to the database are concatentations of arrays of
33  // slices.
34  virtual void Put(ColumnFamilyHandle* column_family, const SliceParts& key,
35  const SliceParts& value);
36  virtual void Put(const SliceParts& key, const SliceParts& value);
37 
38  // Merge "value" with the existing value of "key" in the database.
39  // "key->merge(existing, value)"
40  virtual void Merge(ColumnFamilyHandle* column_family, const Slice& key,
41  const Slice& value) = 0;
42  virtual void Merge(const Slice& key, const Slice& value) = 0;
43 
44  // variant that takes SliceParts
45  virtual void Merge(ColumnFamilyHandle* column_family, const SliceParts& key,
46  const SliceParts& value);
47  virtual void Merge(const SliceParts& key, const SliceParts& value);
48 
49  // If the database contains a mapping for "key", erase it. Else do nothing.
50  virtual void Delete(ColumnFamilyHandle* column_family, const Slice& key) = 0;
51  virtual void Delete(const Slice& key) = 0;
52 
53  // variant that takes SliceParts
54  virtual void Delete(ColumnFamilyHandle* column_family, const SliceParts& key);
55  virtual void Delete(const SliceParts& key);
56 
57  // If the database contains a mapping for "key", erase it. Expects that the
58  // key was not overwritten. Else do nothing.
59  virtual void SingleDelete(ColumnFamilyHandle* column_family,
60  const Slice& key) = 0;
61  virtual void SingleDelete(const Slice& key) = 0;
62 
63  // variant that takes SliceParts
64  virtual void SingleDelete(ColumnFamilyHandle* column_family,
65  const SliceParts& key);
66  virtual void SingleDelete(const SliceParts& key);
67 
68  // Append a blob of arbitrary size to the records in this batch. The blob will
69  // be stored in the transaction log but not in any other file. In particular,
70  // it will not be persisted to the SST files. When iterating over this
71  // WriteBatch, WriteBatch::Handler::LogData will be called with the contents
72  // of the blob as it is encountered. Blobs, puts, deletes, and merges will be
73  // encountered in the same order in thich they were inserted. The blob will
74  // NOT consume sequence number(s) and will NOT increase the count of the batch
75  //
76  // Example application: add timestamps to the transaction log for use in
77  // replication.
78  virtual void PutLogData(const Slice& blob) = 0;
79 
80  // Clear all updates buffered in this batch.
81  virtual void Clear() = 0;
82 
83  // Covert this batch into a WriteBatch. This is an abstracted way of
84  // converting any WriteBatchBase(eg WriteBatchWithIndex) into a basic
85  // WriteBatch.
86  virtual WriteBatch* GetWriteBatch() = 0;
87 
88  // Records the state of the batch for future calls to RollbackToSavePoint().
89  // May be called multiple times to set multiple save points.
90  virtual void SetSavePoint() = 0;
91 
92  // Remove all entries in this batch (Put, Merge, Delete, PutLogData) since the
93  // most recent call to SetSavePoint() and removes the most recent save point.
94  // If there is no previous call to SetSavePoint(), behaves the same as
95  // Clear().
96  virtual Status RollbackToSavePoint() = 0;
97 };
