25 struct TransactionDBOptions {
26  // Specifies the maximum number of keys that can be locked at the same time
27  // per column family.
28  // If the number of locked keys is greater than max_num_locks, transaction
29  // writes (or GetForUpdate) will return an error.
30  // If this value is not positive, no limit will be enforced.
31  int64_t max_num_locks = -1;
32 
33  // Increasing this value will increase the concurrency by dividing the lock
34  // table (per column family) into more sub-tables, each with their own
35  // separate
36  // mutex.
37  size_t num_stripes = 16;
38 
39  // If positive, specifies the default wait timeout in milliseconds when
40  // a transaction attempts to lock a key if not specified by
41  // TransactionOptions::lock_timeout.
42  //
43  // If 0, no waiting is done if a lock cannot instantly be acquired.
44  // If negative, there is no timeout. Not using a timeout is not recommended
45  // as it can lead to deadlocks. Currently, there is no deadlock-detection to
46  // recover
47  // from a deadlock.
48  int64_t transaction_lock_timeout = 1000; // 1 second
49 
50  // If positive, specifies the wait timeout in milliseconds when writing a key
51  // OUTSIDE of a transaction (ie by calling DB::Put(),Merge(),Delete(),Write()
52  // directly).
53  // If 0, no waiting is done if a lock cannot instantly be acquired.
54  // If negative, there is no timeout and will block indefinitely when acquiring
55  // a lock.
56  //
57  // Not using a a timeout can lead to deadlocks. Currently, there
58  // is no deadlock-detection to recover from a deadlock. While DB writes
59  // cannot deadlock with other DB writes, they can deadlock with a transaction.
60  // A negative timeout should only be used if all transactions have an small
61  // expiration set.
62  int64_t default_lock_timeout = 1000; // 1 second
63 
64  // If set, the TransactionDB will use this implemenation of a mutex and
65  // condition variable for all transaction locking instead of the default
66  // mutex/condvar implementation.
67  std::shared_ptr<TransactionDBMutexFactory> custom_mutex_factory;
68 };

70 struct TransactionOptions {
71  // Setting set_snapshot=true is the same as calling
72  // Transaction::SetSnapshot().
73  bool set_snapshot = false;
74 
75 
76  // TODO(agiardullo): TransactionDB does not yet support comparators that allow
77  // two non-equal keys to be equivalent. Ie, cmp->Compare(a,b) should only
78  // return 0 if
79  // a.compare(b) returns 0.
80 
81 
82  // If positive, specifies the wait timeout in milliseconds when
83  // a transaction attempts to lock a key.
84  //
85  // If 0, no waiting is done if a lock cannot instantly be acquired.
86  // If negative, TransactionDBOptions::transaction_lock_timeout will be used.
87  int64_t lock_timeout = -1;
88 
89  // Expiration duration in milliseconds. If non-negative, transactions that
90  // last longer than this many milliseconds will fail to commit. If not set,
91  // a forgotten transaction that is never committed, rolled back, or deleted
92  // will never relinquish any locks it holds. This could prevent keys from
93  // being
94  // written by other writers.
95  int64_t expiration = -1;
96 };
