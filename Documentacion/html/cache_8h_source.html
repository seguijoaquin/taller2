9 // A Cache is an interface that maps keys to values. It has internal
10 // synchronization and may be safely accessed concurrently from
11 // multiple threads. It may automatically evict entries to make room
12 // for new entries. Values have a specified charge against the cache
13 // capacity. For example, a cache where the values are variable
14 // length strings, may use the length of the string as the charge for
15 // the string.
16 //
17 // A builtin cache implementation with a least-recently-used eviction
18 // policy is provided. Clients may use their own implementations if
19 // they want something more sophisticated (like scan-resistance, a
20 // custom eviction policy, variable cache sizing, etc.)

36 // Create a new cache with a fixed size capacity. The cache is sharded
37 // to 2^num_shard_bits shards, by hash of the key. The total capacity
38 // is divided and evenly assigned to each shard.
39 //
40 // The parameter num_shard_bits defaults to 4, and strict_capacity_limit
41 // defaults to false.
42 extern shared_ptr<Cache> NewLRUCache(size_t capacity);
43 extern shared_ptr<Cache> NewLRUCache(size_t capacity, int num_shard_bits);
44 extern shared_ptr<Cache> NewLRUCache(size_t capacity, int num_shard_bits,
45  bool strict_capacity_limit);

47 class Cache {
48  public:
49  Cache() { }
50 
51  // Destroys all existing entries by calling the "deleter"
52  // function that was passed via the Insert() function.
53  //
54  // @See Insert
55  virtual ~Cache();
56 
57  // Opaque handle to an entry stored in the cache.
58  struct Handle { };
59 
60  // Insert a mapping from key->value into the cache and assign it
61  // the specified charge against the total cache capacity.
62  // If strict_capacity_limit is true and cache reaches its full capacity,
63  // return Status::Incomplete.
64  //
65  // If handle is not nullptr, returns a handle that corresponds to the
66  // mapping. The caller must call this->Release(handle) when the returned
67  // mapping is no longer needed. In case of error caller is responsible to
68  // cleanup the value (i.e. calling "deleter").
69  //
70  // If handle is nullptr, it is as if Release is called immediately after
71  // insert. In case of error value will be cleanup.
72  //
73  // When the inserted entry is no longer needed, the key and
74  // value will be passed to "deleter".
75  virtual Status Insert(const Slice& key, void* value, size_t charge,
76  void (*deleter)(const Slice& key, void* value),
77  Handle** handle = nullptr) = 0;
78 
79  // If the cache has no mapping for "key", returns nullptr.
80  //
81  // Else return a handle that corresponds to the mapping. The caller
82  // must call this->Release(handle) when the returned mapping is no
83  // longer needed.
84  virtual Handle* Lookup(const Slice& key) = 0;
85 
86  // Release a mapping returned by a previous Lookup().
87  // REQUIRES: handle must not have been released yet.
88  // REQUIRES: handle must have been returned by a method on *this.
89  virtual void Release(Handle* handle) = 0;
90 
91  // Return the value encapsulated in a handle returned by a
92  // successful Lookup().
93  // REQUIRES: handle must not have been released yet.
94  // REQUIRES: handle must have been returned by a method on *this.
95  virtual void* Value(Handle* handle) = 0;
96 
97  // If the cache contains entry for key, erase it. Note that the
98  // underlying entry will be kept around until all existing handles
99  // to it have been released.
100  virtual void Erase(const Slice& key) = 0;
101 
102  // Return a new numeric id. May be used by multiple clients who are
103  // sharing the same cache to partition the key space. Typically the
104  // client will allocate a new id at startup and prepend the id to
105  // its cache keys.
106  virtual uint64_t NewId() = 0;
107 
108  // sets the maximum configured capacity of the cache. When the new
109  // capacity is less than the old capacity and the existing usage is
110  // greater than new capacity, the implementation will do its best job to
111  // purge the released entries from the cache in order to lower the usage
112  virtual void SetCapacity(size_t capacity) = 0;
113 
114  // Set whether to return error on insertion when cache reaches its full
115  // capacity.
116  virtual void SetStrictCapacityLimit(bool strict_capacity_limit) = 0;
117 
118  // Set whether to return error on insertion when cache reaches its full
119  // capacity.
120  virtual bool HasStrictCapacityLimit() const = 0;
121 
122  // returns the maximum configured capacity of the cache
123  virtual size_t GetCapacity() const = 0;
124 
125  // returns the memory size for the entries residing in the cache.
126  virtual size_t GetUsage() const = 0;
127 
128  // returns the memory size for a specific entry in the cache.
129  virtual size_t GetUsage(Handle* handle) const = 0;
130 
131  // returns the memory size for the entries in use by the system
132  virtual size_t GetPinnedUsage() const = 0;
133 
134  // Call this on shutdown if you want to speed it up. Cache will disown
135  // any underlying data and will not free it on delete. This call will leak
136  // memory - call this only if you're shutting down the process.
137  // Any attempts of using cache after this call will fail terribly.
138  // Always delete the DB object before calling this method!
139  virtual void DisownData() {
140  // default implementation is noop
141  };
142 
143  // Apply callback to all entries in the cache
144  // If thread_safe is true, it will also lock the accesses. Otherwise, it will
145  // access the cache without the lock held
146  virtual void ApplyToAllCacheEntries(void (*callback)(void*, size_t),
147  bool thread_safe) = 0;
148 
149  // Remove all entries.
150  // Prerequisit: no entry is referenced.
151  virtual void EraseUnRefEntries() = 0;
152 
153  private:
154  void LRU_Remove(Handle* e);
155  void LRU_Append(Handle* e);
156  void Unref(Handle* e);
157 
158  // No copying allowed
159  Cache(const Cache&);
160  void operator=(const Cache&);
161 };
