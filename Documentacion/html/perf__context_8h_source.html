16 // A thread local context for gathering performance counter efficiently
17 // and transparently.
18 // Use SetPerfLevel(PerfLevel::kEnableTime) to enable time stats.
19 
20 struct PerfContext {
21 
22  void Reset(); // reset all performance counters to zero
23 
24  std::string ToString(bool exclude_zero_counters = false) const;
25 
26  uint64_t user_key_comparison_count; // total number of user key comparisons
27  uint64_t block_cache_hit_count; // total number of block cache hits
28  uint64_t block_read_count; // total number of block reads (with IO)
29  uint64_t block_read_byte; // total number of bytes from block reads
30  uint64_t block_read_time; // total nanos spent on block reads
31  uint64_t block_checksum_time; // total nanos spent on block checksum
32  uint64_t block_decompress_time; // total nanos spent on block decompression
33  // total number of internal keys skipped over during iteration.
34  // There are several reasons for it:
35  // 1. when calling Next(), the iterator is in the position of the previous
36  // key, so that we'll need to skip it. It means this counter will always
37  // be incremented in Next().
38  // 2. when calling Next(), we need to skip internal entries for the previous
39  // keys that are overwritten.
40  // 3. when calling Next(), Seek() or SeekToFirst(), after previous key
41  // before calling Next(), the seek key in Seek() or the beginning for
42  // SeekToFirst(), there may be one or more deleted keys before the next
43  // valid key that the operation should place the iterator to. We need
44  // to skip both of the tombstone and updates hidden by the tombstones. The
45  // tombstones are not included in this counter, while previous updates
46  // hidden by the tombstones will be included here.
47  // 4. symmetric cases for Prev() and SeekToLast()
48  // We sometimes also skip entries of more recent updates than the snapshot
49  // we read from, but they are not included in this counter.
50  //
51  uint64_t internal_key_skipped_count;
52  // Total number of deletes and single deletes skipped over during iteration
53  // When calling Next(), Seek() or SeekToFirst(), after previous position
54  // before calling Next(), the seek key in Seek() or the beginning for
55  // SeekToFirst(), there may be one or more deleted keys before the next valid
56  // key. Every deleted key is counted once. We don't recount here if there are
57  // still older updates invalidated by the tombstones.
58  //
59  uint64_t internal_delete_skipped_count;
60 
61  uint64_t get_snapshot_time; // total nanos spent on getting snapshot
62  uint64_t get_from_memtable_time; // total nanos spent on querying memtables
63  uint64_t get_from_memtable_count; // number of mem tables queried
64  // total nanos spent after Get() finds a key
65  uint64_t get_post_process_time;
66  uint64_t get_from_output_files_time; // total nanos reading from output files
67  // total nanos spent on seeking memtable
68  uint64_t seek_on_memtable_time;
69  // number of seeks issued on memtable
70  uint64_t seek_on_memtable_count;
71  // total nanos spent on seeking child iters
72  uint64_t seek_child_seek_time;
73  // number of seek issued in child iterators
74  uint64_t seek_child_seek_count;
75  uint64_t seek_min_heap_time; // total nanos spent on the merge heap
76  // total nanos spent on seeking the internal entries
77  uint64_t seek_internal_seek_time;
78  // total nanos spent on iterating internal entries to find the next user entry
79  uint64_t find_next_user_entry_time;
80 
81  // total nanos spent on writing to WAL
82  uint64_t write_wal_time;
83  // total nanos spent on writing to mem tables
84  uint64_t write_memtable_time;
85  // total nanos spent on delaying write
86  uint64_t write_delay_time;
87  // total nanos spent on writing a record, excluding the above three times
88  uint64_t write_pre_and_post_process_time;
89 
90  uint64_t db_mutex_lock_nanos; // time spent on acquiring DB mutex.
91  // Time spent on waiting with a condition variable created with DB mutex.
92  uint64_t db_condition_wait_nanos;
93  // Time spent on merge operator.
94  uint64_t merge_operator_time_nanos;
95 
96  // Time spent on reading index block from block cache or SST file
97  uint64_t read_index_block_nanos;
98  // Time spent on reading filter block from block cache or SST file
99  uint64_t read_filter_block_nanos;
100  // Time spent on creating data block iterator
101  uint64_t new_table_block_iter_nanos;
102  // Time spent on creating a iterator of an SST file.
103  uint64_t new_table_iterator_nanos;
104  // Time spent on seeking a key in data/index blocks
105  uint64_t block_seek_nanos;
106  // Time spent on finding or creating a table reader
107  uint64_t find_table_nanos;
108  // total number of mem table bloom hits
109  uint64_t bloom_memtable_hit_count;
110  // total number of mem table bloom misses
111  uint64_t bloom_memtable_miss_count;
112  // total number of SST table bloom hits
113  uint64_t bloom_sst_hit_count;
114  // total number of SST table bloom misses
115  uint64_t bloom_sst_miss_count;
116 };
