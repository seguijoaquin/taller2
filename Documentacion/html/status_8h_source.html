9 // A Status encapsulates the result of an operation. It may indicate success,
10 // or it may indicate an error with an associated error message.
11 //
12 // Multiple threads can invoke const methods on a Status without
13 // external synchronization, but if any of the threads may call a
14 // non-const method, all threads accessing the same Status must use
15 // external synchronization.

224 inline Status::Status(const Status& s) : code_(s.code_), subcode_(s.subcode_) {
225  state_ = (s.state_ == nullptr) ? nullptr : CopyState(s.state_);
226 }
227 inline Status& Status::operator=(const Status& s) {
228  // The following condition catches both aliasing (when this == &s),
229  // and the common case where both s and *this are ok.
230  if(this != &s) {
231  code_ = s.code_;
232  subcode_ = s.subcode_;
233  delete[] state_;
234  state_ = (s.state_ == nullptr) ? nullptr : CopyState(s.state_);
235  }
236  return *this;
237 }

247 inline Status& Status::operator=(Status&& s)
248 #if !(defined _MSC_VER) || ((defined _MSC_VER) && (_MSC_VER >= 1900))
249 noexcept
250 #endif
251 {
252  if(this != &s) {
253  code_ = std::move(s.code_);
254  s.code_ = kOk;
255  subcode_ = std::move(s.subcode_);
256  s.subcode_ = kNone;
257  delete [] state_;
258  state_ = nullptr;
259  std::swap(state_, s.state_);
260  }
261  return *this;
262 }
