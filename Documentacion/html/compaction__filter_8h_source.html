30 // CompactionFilter allows an application to modify/delete a key-value at
31 // the time of compaction.

33 class CompactionFilter {
34  public:
35  // Context information of a compaction run
36  struct Context {
37  // Does this compaction run include all data files
38  bool is_full_compaction;
39  // Is this compaction requested by the client (true),
40  // or is it occurring as an automatic compaction process
41  bool is_manual_compaction;
42  // Which column family this compaction is for.
43  uint32_t column_family_id;
44  };
45 
46  virtual ~CompactionFilter() {}
47 
48  // The compaction process invokes this
49  // method for kv that is being compacted. A return value
50  // of false indicates that the kv should be preserved in the
51  // output of this compaction run and a return value of true
52  // indicates that this key-value should be removed from the
53  // output of the compaction. The application can inspect
54  // the existing value of the key and make decision based on it.
55  //
56  // Key-Values that are results of merge operation during compaction are not
57  // passed into this function. Currently, when you have a mix of Put()s and
58  // Merge()s on a same key, we only guarantee to process the merge operands
59  // through the compaction filters. Put()s might be processed, or might not.
60  //
61  // When the value is to be preserved, the application has the option
62  // to modify the existing_value and pass it back through new_value.
63  // value_changed needs to be set to true in this case.
64  //
65  // If you use snapshot feature of RocksDB (i.e. call GetSnapshot() API on a
66  // DB* object), CompactionFilter might not be very useful for you. Due to
67  // guarantees we need to maintain, compaction process will not call Filter()
68  // on any keys that were written before the latest snapshot. In other words,
69  // compaction will only call Filter() on keys written after your most recent
70  // call to GetSnapshot(). In most cases, Filter() will not be called very
71  // often. This is something we're fixing. See the discussion at:
72  // https://www.facebook.com/groups/mysqlonrocksdb/permalink/999723240091865/
73  //
74  // If multithreaded compaction is being used *and* a single CompactionFilter
75  // instance was supplied via Options::compaction_filter, this method may be
76  // called from different threads concurrently. The application must ensure
77  // that the call is thread-safe.
78  //
79  // If the CompactionFilter was created by a factory, then it will only ever
80  // be used by a single thread that is doing the compaction run, and this
81  // call does not need to be thread-safe. However, multiple filters may be
82  // in existence and operating concurrently.
83  //
84  // The last paragraph is not true if you set max_subcompactions to more than
85  // 1. In that case, subcompaction from multiple threads may call a single
86  // CompactionFilter concurrently.
87  virtual bool Filter(int level,
88  const Slice& key,
89  const Slice& existing_value,
90  std::string* new_value,
91  bool* value_changed) const = 0;
92 
93  // The compaction process invokes this method on every merge operand. If this
94  // method returns true, the merge operand will be ignored and not written out
95  // in the compaction output
96  //
97  // Note: If you are using a TransactionDB, it is not recommended to implement
98  // FilterMergeOperand(). If a Merge operation is filtered out, TransactionDB
99  // may not realize there is a write conflict and may allow a Transaction to
100  // Commit that should have failed. Instead, it is better to implement any
101  // Merge filtering inside the MergeOperator.
102  virtual bool FilterMergeOperand(int level, const Slice& key,
103  const Slice& operand) const {
104  return false;
105  }
106 
107  // By default, compaction will only call Filter() on keys written after the
108  // most recent call to GetSnapshot(). However, if the compaction filter
109  // overrides IgnoreSnapshots to make it return false, the compaction filter
110  // will be called even if the keys were written before the last snapshot.
111  // This behavior is to be used only when we want to delete a set of keys
112  // irrespective of snapshots. In particular, care should be taken
113  // to understand that the values of thesekeys will change even if we are
114  // using a snapshot.
115  virtual bool IgnoreSnapshots() const { return false; }
116 
117  // Returns a name that identifies this compaction filter.
118  // The name will be printed to LOG file on start up for diagnosis.
119  virtual const char* Name() const = 0;
120 };
