30 struct BackupableDBOptions {
31  // Where to keep the backup files. Has to be different than dbname_
32  // Best to set this to dbname_ + "/backups"
33  // Required
34  std::string backup_dir;
35 
36  // Backup Env object. It will be used for backup file I/O. If it's
37  // nullptr, backups will be written out using DBs Env. If it's
38  // non-nullptr, backup's I/O will be performed using this object.
39  // If you want to have backups on HDFS, use HDFS Env here!
40  // Default: nullptr
41  Env* backup_env;
42 
43  // If share_table_files == true, backup will assume that table files with
44  // same name have the same contents. This enables incremental backups and
45  // avoids unnecessary data copies.
46  // If share_table_files == false, each backup will be on its own and will
47  // not share any data with other backups.
48  // default: true
49  bool share_table_files;
50 
51  // Backup info and error messages will be written to info_log
52  // if non-nullptr.
53  // Default: nullptr
54  Logger* info_log;
55 
56  // If sync == true, we can guarantee you'll get consistent backup even
57  // on a machine crash/reboot. Backup process is slower with sync enabled.
58  // If sync == false, we don't guarantee anything on machine reboot. However,
59  // chances are some of the backups are consistent.
60  // Default: true
61  bool sync;
62 
63  // If true, it will delete whatever backups there are already
64  // Default: false
65  bool destroy_old_data;
66 
67  // If false, we won't backup log files. This option can be useful for backing
68  // up in-memory databases where log file are persisted, but table files are in
69  // memory.
70  // Default: true
71  bool backup_log_files;
72 
73  // Max bytes that can be transferred in a second during backup.
74  // If 0, go as fast as you can
75  // Default: 0
76  uint64_t backup_rate_limit;
77 
78  // Max bytes that can be transferred in a second during restore.
79  // If 0, go as fast as you can
80  // Default: 0
81  uint64_t restore_rate_limit;
82 
83  // Only used if share_table_files is set to true. If true, will consider that
84  // backups can come from different databases, hence a sst is not uniquely
85  // identifed by its name, but by the triple (file name, crc32, file length)
86  // Default: false
87  // Note: this is an experimental option, and you'll need to set it manually
88  // *turn it on only if you know what you're doing*
89  bool share_files_with_checksum;
90 
91  // Up to this many background threads will copy files for CreateNewBackup()
92  // and RestoreDBFromBackup()
93  // Default: 1
94  int max_background_operations;
95 
96  // During backup user can get callback every time next
97  // callback_trigger_interval_size bytes being copied.
98  // Default: 4194304
99  uint64_t callback_trigger_interval_size;
100 
101  void Dump(Logger* logger) const;
102 
103  explicit BackupableDBOptions(
104  const std::string& _backup_dir, Env* _backup_env = nullptr,
105  bool _share_table_files = true, Logger* _info_log = nullptr,
106  bool _sync = true, bool _destroy_old_data = false,
107  bool _backup_log_files = true, uint64_t _backup_rate_limit = 0,
108  uint64_t _restore_rate_limit = 0, int _max_background_operations = 1,
109  uint64_t _callback_trigger_interval_size = 4 * 1024 * 1024)
110  : backup_dir(_backup_dir),
111  backup_env(_backup_env),
112  share_table_files(_share_table_files),
113  info_log(_info_log),
114  sync(_sync),
115  destroy_old_data(_destroy_old_data),
116  backup_log_files(_backup_log_files),
117  backup_rate_limit(_backup_rate_limit),
118  restore_rate_limit(_restore_rate_limit),
119  share_files_with_checksum(false),
120  max_background_operations(_max_background_operations),
121  callback_trigger_interval_size(_callback_trigger_interval_size) {
122  assert(share_table_files || !share_files_with_checksum);
123  }
124 };

258 // Stack your DB with BackupableDB to be able to backup the DB
259 class BackupableDB : public StackableDB {
260  public:
261  // BackupableDBOptions have to be the same as the ones used in a previous
262  // incarnation of the DB
263  //
264  // BackupableDB ownes the pointer `DB* db` now. You should not delete it or
265  // use it after the invocation of BackupableDB
266  BackupableDB(DB* db, const BackupableDBOptions& options);
267  virtual ~BackupableDB();
268 
269  // Captures the state of the database in the latest backup
270  // NOT a thread safe call
271  Status CreateNewBackup(bool flush_before_backup = false);
272  // Returns info about backups in backup_info
273  void GetBackupInfo(std::vector<BackupInfo>* backup_info);
274  // Returns info about corrupt backups in corrupt_backups
275  void GetCorruptedBackups(std::vector<BackupID>* corrupt_backup_ids);
276  // deletes old backups, keeping latest num_backups_to_keep alive
277  Status PurgeOldBackups(uint32_t num_backups_to_keep);
278  // deletes a specific backup
279  Status DeleteBackup(BackupID backup_id);
280  // Call this from another thread if you want to stop the backup
281  // that is currently happening. It will return immediatelly, will
282  // not wait for the backup to stop.
283  // The backup will stop ASAP and the call to CreateNewBackup will
284  // return Status::Incomplete(). It will not clean up after itself, but
285  // the state will remain consistent. The state will be cleaned up
286  // next time you create BackupableDB or RestoreBackupableDB.
287  void StopBackup();
288 
289  // Will delete all the files we don't need anymore
290  // It will do the full scan of the files/ directory and delete all the
291  // files that are not referenced.
292  Status GarbageCollect();
293 
294  private:
295  BackupEngine* backup_engine_;
296  Status status_;
297 };

311  // IMPORTANT -- if options_.share_table_files == true and you restore DB

314  //
315  // Example: Let's say you have backups 1, 2, 3, 4, 5 and you restore 3.
316  // If you add new data to the DB and try creating a new backup now, the

318  // If you want to create new backup, you will first have to delete backups 4
319  // and 5.

321  const std::string& wal_dir,
322  const RestoreOptions& restore_options =
323  RestoreOptions());
