72 // A WriteBatchWithIndex with a binary searchable index built for all the keys
73 // inserted.
74 // In Put(), Merge() Delete(), or SingleDelete(), the same function of the
75 // wrapped will be called. At the same time, indexes will be built.
76 // By calling GetWriteBatch(), a user will get the WriteBatch for the data
77 // they inserted, which can be used for DB::Write().
78 // A user can call NewIterator() to create an iterator.
79 class WriteBatchWithIndex : public WriteBatchBase {
80  public:
81  // backup_index_comparator: the backup comparator used to compare keys
82  // within the same column family, if column family is not given in the
83  // interface, or we can't find a column family from the column family handle
84  // passed in, backup_index_comparator will be used for the column family.
85  // reserved_bytes: reserved bytes in underlying WriteBatch
86  // overwrite_key: if true, overwrite the key in the index when inserting
87  // the same key as previously, so iterator will never
88  // show two entries with the same key.
89  explicit WriteBatchWithIndex(
90  const Comparator* backup_index_comparator = BytewiseComparator(),
91  size_t reserved_bytes = 0, bool overwrite_key = false);
92  virtual ~WriteBatchWithIndex();
93 
94  using WriteBatchBase::Put;
95  void Put(ColumnFamilyHandle* column_family, const Slice& key,
96  const Slice& value) override;
97 
98  void Put(const Slice& key, const Slice& value) override;
99 
100  using WriteBatchBase::Merge;
101  void Merge(ColumnFamilyHandle* column_family, const Slice& key,
102  const Slice& value) override;
103 
104  void Merge(const Slice& key, const Slice& value) override;
105 
106  using WriteBatchBase::Delete;
107  void Delete(ColumnFamilyHandle* column_family, const Slice& key) override;
108  void Delete(const Slice& key) override;
109 
110  using WriteBatchBase::SingleDelete;
111  void SingleDelete(ColumnFamilyHandle* column_family,
112  const Slice& key) override;
113  void SingleDelete(const Slice& key) override;
114 
115  using WriteBatchBase::PutLogData;
116  void PutLogData(const Slice& blob) override;
117 
118  using WriteBatchBase::Clear;
119  void Clear() override;
120 
121  using WriteBatchBase::GetWriteBatch;
122  WriteBatch* GetWriteBatch() override;
123 
124  // Create an iterator of a column family. User can call iterator.Seek() to
125  // search to the next entry of or after a key. Keys will be iterated in the
126  // order given by index_comparator. For multiple updates on the same key,
127  // each update will be returned as a separate entry, in the order of update
128  // time.
129  //
130  // The returned iterator should be deleted by the caller.
131  WBWIIterator* NewIterator(ColumnFamilyHandle* column_family);
132  // Create an iterator of the default column family.
133  WBWIIterator* NewIterator();
134 
135  // Will create a new Iterator that will use WBWIIterator as a delta and
136  // base_iterator as base.
137  //
138  // This function is only supported if the WriteBatchWithIndex was
139  // constructed with overwrite_key=true.
140  //
141  // The returned iterator should be deleted by the caller.
142  // The base_iterator is now 'owned' by the returned iterator. Deleting the
143  // returned iterator will also delete the base_iterator.
144  Iterator* NewIteratorWithBase(ColumnFamilyHandle* column_family,
145  Iterator* base_iterator);
146  // default column family
147  Iterator* NewIteratorWithBase(Iterator* base_iterator);
148 
149  // Similar to DB::Get() but will only read the key from this batch.
150  // If the batch does not have enough data to resolve Merge operations,
151  // MergeInProgress status may be returned.
152  Status GetFromBatch(ColumnFamilyHandle* column_family,
153  const DBOptions& options, const Slice& key,
154  std::string* value);
155 
156  // Similar to previous function but does not require a column_family.
157  // Note: An InvalidArgument status will be returned if there are any Merge
158  // operators for this key. Use previous method instead.
159  Status GetFromBatch(const DBOptions& options, const Slice& key,
160  std::string* value) {
161  return GetFromBatch(nullptr, options, key, value);
162  }
163 
164  // Similar to DB::Get() but will also read writes from this batch.
165  //
166  // This function will query both this batch and the DB and then merge
167  // the results using the DB's merge operator (if the batch contains any
168  // merge requests).
169  //
170  // Setting read_options.snapshot will affect what is read from the DB
171  // but will NOT change which keys are read from the batch (the keys in
172  // this batch do not yet belong to any snapshot and will be fetched
173  // regardless).
174  Status GetFromBatchAndDB(DB* db, const ReadOptions& read_options,
175  const Slice& key, std::string* value);
176  Status GetFromBatchAndDB(DB* db, const ReadOptions& read_options,
177  ColumnFamilyHandle* column_family, const Slice& key,
178  std::string* value);
179 
180  // Records the state of the batch for future calls to RollbackToSavePoint().
181  // May be called multiple times to set multiple save points.
182  void SetSavePoint() override;
183 
184  // Remove all entries in this batch (Put, Merge, Delete, SingleDelete,
185  // PutLogData) since the most recent call to SetSavePoint() and removes the
186  // most recent save point.
187  // If there is no previous call to SetSavePoint(), behaves the same as
188  // Clear().
189  //
190  // Calling RollbackToSavePoint invalidates any open iterators on this batch.
191  //
192  // Returns Status::OK() on success,
193  // Status::NotFound() if no previous call to SetSavePoint(),
194  // or other Status on corruption.
195  Status RollbackToSavePoint() override;
196 
197  private:
198  struct Rep;
199  Rep* rep;
200 };
