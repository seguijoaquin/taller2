5 // Currently we support two types of tables: plain table and block-based table.
6 // 1. Block-based table: this is the default table type that we inherited from
7 // LevelDB, which was designed for storing data in hard disk or flash
8 // device.
9 // 2. Plain table: it is one of RocksDB's SST file format optimized
10 // for low query latency on pure-memory or really low-latency media.
11 //
12 // A tutorial of rocksdb table formats is available here:
13 // https://github.com/facebook/rocksdb/wiki/A-Tutorial-of-RocksDB-SST-formats
14 //
15 // Example code is also available
16 // https://github.com/facebook/rocksdb/wiki/A-Tutorial-of-RocksDB-SST-formats#wiki-examples

53  // which provides a configurable way to determine when to flush a block in
54  // the block based tables. If not set, table builder will use the default

58 
59  // TODO(kailiu) Temporarily disable this feature by making the default value
60  // to be false.
61  //
62  // Indicating if we'd put index/filter blocks to the block cache.
63  // If not specified, each "table reader" object will pre-load index/filter
64  // block during table initialization.
65  bool cache_index_and_filter_blocks = false;
66 
67  // if cache_index_and_filter_blocks is true and the below is true, then
68  // filter and index blocks are stored in the cache, but a reference is
69  // held in the "table reader" object so the blocks are pinned and only
70  // evicted from cache when the table reader is freed.
71  bool pin_l0_filter_and_index_blocks_in_cache = false;

73  // The index type that will be used for this table.
74  enum IndexType : char {
75  // A space efficient index block that is optimized for
76  // binary-search-based index.
77  kBinarySearch,
78 
79  // The hash index, if enabled, will do the hash lookup when
80  // `Options.prefix_extractor` is provided.
81  kHashSearch,
82  };
83 
84  IndexType index_type = kBinarySearch;
85 
86  // Influence the behavior when kHashSearch is used.
87  // if false, stores a precise prefix to block range mapping
88  // if true, does not store prefix and allows prefix hash collision
89  // (less memory consumption)
90  bool hash_index_allow_collision = true;

93  // protected with this checksum type. Old table files will still be readable,
94  // even though they have different checksum type.
95  ChecksumType checksum = kCRC32c;
96 
97  // Disable block cache. If this is set to true,
98  // then no block cache should be used, and the block_cache should
99  // point to a nullptr object.
100  bool no_block_cache = false;
101 
102  // If non-NULL use the specified cache for blocks.
103  // If NULL, rocksdb will automatically create and use an 8MB internal cache.
104  std::shared_ptr<Cache> block_cache = nullptr;
105 
106  // If non-NULL use the specified cache for compressed blocks.
107  // If NULL, rocksdb will not use a compressed block cache.
108  std::shared_ptr<Cache> block_cache_compressed = nullptr;
109 
110  // Approximate size of user data packed per block. Note that the
111  // block size specified here corresponds to uncompressed data. The
112  // actual size of the unit read from disk may be smaller if
113  // compression is enabled. This parameter can be changed dynamically.
114  size_t block_size = 4 * 1024;
115 
116  // This is used to close a block before it reaches the configured
117  // 'block_size'. If the percentage of free space in the current block is less
118  // than this specified number and adding a new record to the block will
119  // exceed the configured block size, then this block will be closed and the
120  // new record will be written to the next block.
121  int block_size_deviation = 10;
122 
123  // Number of keys between restart points for delta encoding of keys.
124  // This parameter can be changed dynamically. Most clients should
125  // leave this parameter alone. The minimum value allowed is 1. Any smaller
126  // value will be silently overwritten with 1.
127  int block_restart_interval = 16;
128 
129  // Same as block_restart_interval but used for the index block.
130  int index_block_restart_interval = 1;
131 
132  // Use delta encoding to compress keys in blocks.
133  // Iterator::PinData() requires this option to be disabled.
134  //
135  // Default: true
136  bool use_delta_encoding = true;
137 
138  // If non-nullptr, use the specified filter policy to reduce disk reads.
139  // Many applications will benefit from passing the result of
140  // NewBloomFilterPolicy() here.
141  std::shared_ptr<const FilterPolicy> filter_policy = nullptr;
142 
143  // If true, place whole keys in the filter (not just prefixes).
144  // This must generally be true for gets to be efficient.
145  bool whole_key_filtering = true;
146 
147  // If true, block will not be explicitly flushed to disk during building
148  // a SstTable. Instead, buffer in WritableFileWriter will take
149  // care of the flushing when it is full.
150  //
151  // On Windows, this option helps a lot when unbuffered I/O
152  // (allow_os_buffer = false) is used, since it avoids small
153  // unbuffered disk write.
154  //
155  // User may also adjust writable_file_max_buffer_size to optimize disk I/O
156  // size.
157  //
158  // Default: false
159  bool skip_table_builder_flush = false;
160 
161  // We currently have three versions:
162  // 0 -- This version is currently written out by all RocksDB's versions by
163  // default. Can be read by really old RocksDB's. Doesn't support changing
164  // checksum (default is CRC32).
165  // 1 -- Can be read by RocksDB's versions since 3.0. Supports non-default
166  // checksum, like xxHash. It is written by RocksDB when
167  // BlockBasedTableOptions::checksum is something other than kCRC32c. (version
168  // 0 is silently upconverted)
169  // 2 -- Can be read by RocksDB's versions since 3.10. Changes the way we
170  // encode compressed blocks with LZ4, BZip2 and Zlib compression. If you
171  // don't plan to run RocksDB before version 3.10, you should probably use
172  // this.
173  // This option only affects newly written tables. When reading exising tables,
174  // the information about version is read from the footer.
175  uint32_t format_version = 2;

194 enum EncodingType : char {
195  // Always write full keys without any special encoding.
196  kPlain,
197  // Find opportunity to write the same prefix once for multiple rows.
198  // In some cases, when a key follows a previous key with the same prefix,
199  // instead of writing out the full key, it just writes out the size of the
200  // shared prefix, as well as other bytes, to save some bytes.
201  //
202  // When using this option, the user is required to use the same prefix
203  // extractor to make sure the same prefix will be extracted from the same key.
204  // The Name() value of the prefix extractor will be stored in the file. When
205  // reopening the file, the name of the options.prefix_extractor given will be
206  // bitwise compared to the prefix extractors stored in the file. An error
207  // will be returned if the two don't match.
208  kPrefix,
209 };

221 struct PlainTableOptions {
222  // @user_key_len: plain table has optimization for fix-sized keys, which can
223  // be specified via user_key_len. Alternatively, you can pass
224  // `kPlainTableVariableLength` if your keys have variable
225  // lengths.
226  uint32_t user_key_len = kPlainTableVariableLength;
227 
228  // @bloom_bits_per_key: the number of bits used for bloom filer per prefix.
229  // You may disable it by passing a zero.
230  int bloom_bits_per_key = 10;
231 
232  // @hash_table_ratio: the desired utilization of the hash table used for
233  // prefix hashing.
234  // hash_table_ratio = number of prefixes / #buckets in the
235  // hash table
236  double hash_table_ratio = 0.75;
237 
238  // @index_sparseness: inside each prefix, need to build one index record for
239  // how many keys for binary search inside each hash bucket.
240  // For encoding type kPrefix, the value will be used when
241  // writing to determine an interval to rewrite the full
242  // key. It will also be used as a suggestion and satisfied
243  // when possible.
244  size_t index_sparseness = 16;
245 
246  // @huge_page_tlb_size: if <=0, allocate hash indexes and blooms from malloc.
247  // Otherwise from huge page TLB. The user needs to
248  // reserve huge pages for it to be allocated, like:
249  // sysctl -w vm.nr_hugepages=20
250  // See linux doc Documentation/vm/hugetlbpage.txt
251  size_t huge_page_tlb_size = 0;
252 
253  // @encoding_type: how to encode the keys. See enum EncodingType above for
254  // the choices. The value will determine how to encode keys
255  // when writing to a new SST file. This value will be stored
256  // inside the SST file which will be used when reading from
257  // the file, which makes it possible for users to choose
258  // different encoding type when reopening a DB. Files with
259  // different encoding types can co-exist in the same DB and
260  // can be read.
261  EncodingType encoding_type = kPlain;
262 
263  // @full_scan_mode: mode for reading the whole file one record by one without
264  // using the index.
265  bool full_scan_mode = false;
266 
267  // @store_index_in_file: compute plain table index and bloom filter during
268  // file building and store it in file. When reading
269  // file, index will be mmaped instead of recomputation.
270  bool store_index_in_file = false;
271 };

282 struct CuckooTablePropertyNames {
283  // The key that is used to fill empty buckets.
284  static const std::string kEmptyKey;
285  // Fixed length of value.
286  static const std::string kValueLength;
287  // Number of hash functions used in Cuckoo Hash.
288  static const std::string kNumHashFunc;
289  // It denotes the number of buckets in a Cuckoo Block. Given a key and a
290  // particular hash function, a Cuckoo Block is a set of consecutive buckets,
291  // where starting bucket id is given by the hash function on the key. In case
292  // of a collision during inserting the key, the builder tries to insert the
293  // key in other locations of the cuckoo block before using the next hash
294  // function. This reduces cache miss during read operation in case of
295  // collision.
296  static const std::string kCuckooBlockSize;
297  // Size of the hash table. Use this number to compute the modulo of hash
298  // function. The actual number of buckets will be kMaxHashTableSize +
299  // kCuckooBlockSize - 1. The last kCuckooBlockSize-1 buckets are used to
300  // accommodate the Cuckoo Block from end of hash table, due to cache friendly
301  // implementation.
302  static const std::string kHashTableSize;
303  // Denotes if the key sorted in the file is Internal Key (if false)
304  // or User Key only (if true).
305  static const std::string kIsLastLevel;
306  // Indicate if using identity function for the first hash function.
307  static const std::string kIdentityAsFirstHash;
308  // Indicate if using module or bit and to calculate hash value
309  static const std::string kUseModuleHash;
310  // Fixed user key length
311  static const std::string kUserKeyLength;
312 };

314 struct CuckooTableOptions {
315  // Determines the utilization of hash tables. Smaller values
316  // result in larger hash tables with fewer collisions.
317  double hash_table_ratio = 0.9;
318  // A property used by builder to determine the depth to go to
319  // to search for a path to displace elements in case of
320  // collision. See Builder.MakeSpaceForKey method. Higher
321  // values result in more efficient hash tables with fewer
322  // lookups but take more time to build.
323  uint32_t max_search_depth = 100;
324  // In case of collision while inserting, the builder
325  // attempts to insert in the next cuckoo_block_size
326  // locations before skipping over to the next Cuckoo hash
327  // function. This makes lookups more cache friendly in case
328  // of collisions.
329  uint32_t cuckoo_block_size = 5;
330  // If this option is enabled, user key is treated as uint64_t and its value
331  // is used as hash value directly. This option changes builder's behavior.
332  // Reader ignore this option and behave according to what specified in table
333  // property.
334  bool identity_as_first_hash = false;
335  // If this option is set to true, module is used during hash calculation.
336  // This often yields better space efficiency at the cost of performance.
337  // If this optino is set to false, # of entries in table is constrained to be
338  // power of two, and bit and is used to calculate hash, which is faster in
339  // general.
340  bool use_module_hash = true;
341 };

351 // A base class for table factories.
352 class TableFactory {
353  public:
354  virtual ~TableFactory() {}
355 
356  // The type of the table.
357  //
358  // The client of this package should switch to a new name whenever
359  // the table format implementation changes.
360  //
361  // Names starting with "rocksdb." are reserved and should not be used
362  // by any clients of this package.
363  virtual const char* Name() const = 0;
364 
365  // Returns a Table object table that can fetch data from file specified
366  // in parameter file. It's the caller's responsibility to make sure
367  // file is in the correct format.
368  //
369  // NewTableReader() is called in three places:
370  // (1) TableCache::FindTable() calls the function when table cache miss
371  // and cache the table object returned.
372  // (2) SstFileReader (for SST Dump) opens the table and dump the table
373  // contents using the interator of the table.
374  // (3) DBImpl::AddFile() calls this function to read the contents of
375  // the sst file it's attempting to add
376  //
377  // table_reader_options is a TableReaderOptions which contain all the
378  // needed parameters and configuration to open the table.
379  // file is a file handler to handle the file for the table.
380  // file_size is the physical file size of the file.
381  // table_reader is the output table reader.
382  virtual Status NewTableReader(
383  const TableReaderOptions& table_reader_options,
384  unique_ptr<RandomAccessFileReader>&& file, uint64_t file_size,
385  unique_ptr<TableReader>* table_reader) const = 0;
386 
387  // Return a table builder to write to a file for this table type.
388  //
389  // It is called in several places:
390  // (1) When flushing memtable to a level-0 output file, it creates a table
391  // builder (In DBImpl::WriteLevel0Table(), by calling BuildTable())
392  // (2) During compaction, it gets the builder for writing compaction output
393  // files in DBImpl::OpenCompactionOutputFile().
394  // (3) When recovering from transaction logs, it creates a table builder to
395  // write to a level-0 output file (In DBImpl::WriteLevel0TableForRecovery,
396  // by calling BuildTable())
397  // (4) When running Repairer, it creates a table builder to convert logs to
398  // SST files (In Repairer::ConvertLogToTable() by calling BuildTable())
399  //
400  // ImmutableCFOptions is a subset of Options that can not be altered.
401  // Multiple configured can be acceseed from there, including and not limited
402  // to compression options. file is a handle of a writable file.
403  // It is the caller's responsibility to keep the file open and close the file
404  // after closing the table builder. compression_type is the compression type
405  // to use in this table.
406  virtual TableBuilder* NewTableBuilder(
407  const TableBuilderOptions& table_builder_options,
408  uint32_t column_family_id, WritableFileWriter* file) const = 0;
409 
410  // Sanitizes the specified DB Options and ColumnFamilyOptions.
411  //
412  // If the function cannot find a way to sanitize the input DB Options,
413  // a non-ok Status will be returned.
414  virtual Status SanitizeOptions(
415  const DBOptions& db_opts,
416  const ColumnFamilyOptions& cf_opts) const = 0;
417 
418  // Return a string that contains printable format of table configurations.
419  // RocksDB prints configurations at DB Open().
420  virtual std::string GetPrintableTableOptions() const = 0;
421 
422  // Returns the raw pointer of the table options that is used by this
423  // TableFactory, or nullptr if this function is not supported.
424  // Since the return value is a raw pointer, the TableFactory owns the
425  // pointer and the caller should not delete the pointer.
426  //
427  // In certan case, it is desirable to alter the underlying options when the
428  // TableFactory is not used by any open DB by casting the returned pointer
429  // to the right class. For instance, if BlockBasedTableFactory is used,
430  // then the pointer can be casted to BlockBasedTableOptions.
431  //
432  // Note that changing the underlying TableFactory options while the
433  // TableFactory is currently used by any open DB is undefined behavior.
434  // Developers should use DB::SetOption() instead to dynamically change
435  // options while the DB is open.
436  virtual void* GetOptions() { return nullptr; }
437 };
