20 // The Merge Operator
21 //
22 // Essentially, a MergeOperator specifies the SEMANTICS of a merge, which only
23 // client knows. It could be numeric addition, list append, string
24 // concatenation, edit data structure, ... , anything.
25 // The library, on the other hand, is concerned with the exercise of this
26 // interface, at the right time (during get, iteration, compaction...)
27 //
28 // To use merge, the client needs to provide an object implementing one of
29 // the following interfaces:
30 // a) AssociativeMergeOperator - for most simple semantics (always take
31 // two values, and merge them into one value, which is then put back
32 // into rocksdb); numeric addition and string concatenation are examples;
33 //
34 // b) MergeOperator - the generic class for all the more abstract / complex
35 // operations; one method (FullMerge) to merge a Put/Delete value with a
36 // merge operand; and another method (PartialMerge) that merges multiple
37 // operands together. this is especially useful if your key values have
38 // complex structures but you would still like to support client-specific
39 // incremental updates.
40 //
41 // AssociativeMergeOperator is simpler to implement. MergeOperator is simply
42 // more powerful.
43 //
44 // Refer to rocksdb-merge wiki for more details and example implementations.
45 //
46 class MergeOperator {
47  public:
48  virtual ~MergeOperator() {}
49 
50  // Gives the client a way to express the read -> modify -> write semantics
51  // key: (IN) The key that's associated with this merge operation.
52  // Client could multiplex the merge operator based on it
53  // if the key space is partitioned and different subspaces
54  // refer to different types of data which have different
55  // merge operation semantics
56  // existing: (IN) null indicates that the key does not exist before this op
57  // operand_list:(IN) the sequence of merge operations to apply, front() first.
58  // new_value:(OUT) Client is responsible for filling the merge result here.
59  // The string that new_value is pointing to will be empty.
60  // logger: (IN) Client could use this to log errors during merge.
61  //
62  // Return true on success.
63  // All values passed in will be client-specific values. So if this method
64  // returns false, it is because client specified bad data or there was
65  // internal corruption. This will be treated as an error by the library.
66  //
67  // Also make use of the *logger for error messages.
68  virtual bool FullMerge(const Slice& key,
69  const Slice* existing_value,
70  const std::deque<std::string>& operand_list,
71  std::string* new_value,
72  Logger* logger) const = 0;
73 
74  // This function performs merge(left_op, right_op)
75  // when both the operands are themselves merge operation types
76  // that you would have passed to a DB::Merge() call in the same order
77  // (i.e.: DB::Merge(key,left_op), followed by DB::Merge(key,right_op)).
78  //
79  // PartialMerge should combine them into a single merge operation that is
80  // saved into *new_value, and then it should return true.
81  // *new_value should be constructed such that a call to
82  // DB::Merge(key, *new_value) would yield the same result as a call
83  // to DB::Merge(key, left_op) followed by DB::Merge(key, right_op).
84  //
85  // The string that new_value is pointing to will be empty.
86  //
87  // The default implementation of PartialMergeMulti will use this function
88  // as a helper, for backward compatibility. Any successor class of
89  // MergeOperator should either implement PartialMerge or PartialMergeMulti,
90  // although implementing PartialMergeMulti is suggested as it is in general
91  // more effective to merge multiple operands at a time instead of two
92  // operands at a time.
93  //
94  // If it is impossible or infeasible to combine the two operations,
95  // leave new_value unchanged and return false. The library will
96  // internally keep track of the operations, and apply them in the
97  // correct order once a base-value (a Put/Delete/End-of-Database) is seen.
98  //
99  // TODO: Presently there is no way to differentiate between error/corruption
100  // and simply "return false". For now, the client should simply return
101  // false in any case it cannot perform partial-merge, regardless of reason.
102  // If there is corruption in the data, handle it in the FullMerge() function,
103  // and return false there. The default implementation of PartialMerge will
104  // always return false.
105  virtual bool PartialMerge(const Slice& key, const Slice& left_operand,
106  const Slice& right_operand, std::string* new_value,
107  Logger* logger) const {
108  return false;
109  }
110 
111  // This function performs merge when all the operands are themselves merge
112  // operation types that you would have passed to a DB::Merge() call in the
113  // same order (front() first)
114  // (i.e. DB::Merge(key, operand_list[0]), followed by
115  // DB::Merge(key, operand_list[1]), ...)
116  //
117  // PartialMergeMulti should combine them into a single merge operation that is
118  // saved into *new_value, and then it should return true. *new_value should
119  // be constructed such that a call to DB::Merge(key, *new_value) would yield
120  // the same result as subquential individual calls to DB::Merge(key, operand)
121  // for each operand in operand_list from front() to back().
122  //
123  // The string that new_value is pointing to will be empty.
124  //
125  // The PartialMergeMulti function will be called only when the list of
126  // operands are long enough. The minimum amount of operands that will be
127  // passed to the function are specified by the "min_partial_merge_operands"
128  // option.
129  //
130  // In the default implementation, PartialMergeMulti will invoke PartialMerge
131  // multiple times, where each time it only merges two operands. Developers
132  // should either implement PartialMergeMulti, or implement PartialMerge which
133  // is served as the helper function of the default PartialMergeMulti.
134  virtual bool PartialMergeMulti(const Slice& key,
135  const std::deque<Slice>& operand_list,
136  std::string* new_value, Logger* logger) const;
137 
138  // The name of the MergeOperator. Used to check for MergeOperator
139  // mismatches (i.e., a DB created with one MergeOperator is
140  // accessed using a different MergeOperator)
141  // TODO: the name is currently not stored persistently and thus
142  // no checking is enforced. Client is responsible for providing
143  // consistent MergeOperator between DB opens.
144  virtual const char* Name() const = 0;
145 };
