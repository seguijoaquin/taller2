41 // A structure that describes the current status of a thread.
42 // The status of active threads can be fetched using
43 // rocksdb::GetThreadList().
44 struct ThreadStatus {
45  // The type of a thread.
46  enum ThreadType : int {
47  HIGH_PRIORITY = 0, // RocksDB BG thread in high-pri thread pool
48  LOW_PRIORITY, // RocksDB BG thread in low-pri thread pool
49  USER, // User thread (Non-RocksDB BG thread)
50  NUM_THREAD_TYPES
51  };
52 
53  // The type used to refer to a thread operation.
54  // A thread operation describes high-level action of a thread.
55  // Examples include compaction and flush.
56  enum OperationType : int {
57  OP_UNKNOWN = 0,
58  OP_COMPACTION,
59  OP_FLUSH,
60  NUM_OP_TYPES
61  };
62 
63  enum OperationStage : int {
64  STAGE_UNKNOWN = 0,
65  STAGE_FLUSH_RUN,
66  STAGE_FLUSH_WRITE_L0,
67  STAGE_COMPACTION_PREPARE,
68  STAGE_COMPACTION_RUN,
69  STAGE_COMPACTION_PROCESS_KV,
70  STAGE_COMPACTION_INSTALL,
71  STAGE_COMPACTION_SYNC_FILE,
72  STAGE_PICK_MEMTABLES_TO_FLUSH,
73  STAGE_MEMTABLE_ROLLBACK,
74  STAGE_MEMTABLE_INSTALL_FLUSH_RESULTS,
75  NUM_OP_STAGES
76  };
77 
78  enum CompactionPropertyType : int {
79  COMPACTION_JOB_ID = 0,
80  COMPACTION_INPUT_OUTPUT_LEVEL,
81  COMPACTION_PROP_FLAGS,
82  COMPACTION_TOTAL_INPUT_BYTES,
83  COMPACTION_BYTES_READ,
84  COMPACTION_BYTES_WRITTEN,
85  NUM_COMPACTION_PROPERTIES
86  };
87 
88  enum FlushPropertyType : int {
89  FLUSH_JOB_ID = 0,
90  FLUSH_BYTES_MEMTABLES,
91  FLUSH_BYTES_WRITTEN,
92  NUM_FLUSH_PROPERTIES
93  };
94 
95  // The maximum number of properties of an operation.
96  // This number should be set to the biggest NUM_XXX_PROPERTIES.
97  static const int kNumOperationProperties =
98  constexpr_max<NUM_COMPACTION_PROPERTIES, NUM_FLUSH_PROPERTIES>::result;
99 
100  // The type used to refer to a thread state.
101  // A state describes lower-level action of a thread
102  // such as reading / writing a file or waiting for a mutex.
103  enum StateType : int {
104  STATE_UNKNOWN = 0,
105  STATE_MUTEX_WAIT = 1,
106  NUM_STATE_TYPES
107  };
108 
109  ThreadStatus(const uint64_t _id,
110  const ThreadType _thread_type,
111  const std::string& _db_name,
112  const std::string& _cf_name,
113  const OperationType _operation_type,
114  const uint64_t _op_elapsed_micros,
115  const OperationStage _operation_stage,
116  const uint64_t _op_props[],
117  const StateType _state_type) :
118  thread_id(_id), thread_type(_thread_type),
119  db_name(_db_name),
120  cf_name(_cf_name),
121  operation_type(_operation_type),
122  op_elapsed_micros(_op_elapsed_micros),
123  operation_stage(_operation_stage),
124  state_type(_state_type) {
125  for (int i = 0; i < kNumOperationProperties; ++i) {
126  op_properties[i] = _op_props[i];
127  }
128  }
129 
130  // An unique ID for the thread.
131  const uint64_t thread_id;
132 
133  // The type of the thread, it could be HIGH_PRIORITY,
134  // LOW_PRIORITY, and USER
135  const ThreadType thread_type;
136 
137  // The name of the DB instance where the thread is currently
138  // involved with. It would be set to empty string if the thread
139  // does not involve in any DB operation.
140  const std::string db_name;
141 
142  // The name of the column family where the thread is currently
143  // It would be set to empty string if the thread does not involve
144  // in any column family.
145  const std::string cf_name;
146 
147  // The operation (high-level action) that the current thread is involved.
148  const OperationType operation_type;
149 
150  // The elapsed time in micros of the current thread operation.
151  const uint64_t op_elapsed_micros;
152 
153  // An integer showing the current stage where the thread is involved
154  // in the current operation.
155  const OperationStage operation_stage;
156 
157  // A list of properties that describe some details about the current
158  // operation. Same field in op_properties[] might have different
159  // meanings for different operations.
160  uint64_t op_properties[kNumOperationProperties];
161 
162  // The state (lower-level action) that the current thread is involved.
163  const StateType state_type;
164 
165  // The followings are a set of utility functions for interpreting
166  // the information of ThreadStatus
167 
168  static const std::string& GetThreadTypeName(ThreadType thread_type);
169 
170  // Obtain the name of an operation given its type.
171  static const std::string& GetOperationName(OperationType op_type);
172 
173  static const std::string MicrosToString(uint64_t op_elapsed_time);
174 
175  // Obtain a human-readable string describing the specified operation stage.
176  static const std::string& GetOperationStageName(
177  OperationStage stage);
178 
179  // Obtain the name of the "i"th operation property of the
180  // specified operation.
181  static const std::string& GetOperationPropertyName(
182  OperationType op_type, int i);
183 
184  // Translate the "i"th property of the specified operation given
185  // a property value.
186  static std::map<std::string, uint64_t>
187  InterpretOperationProperties(
188  OperationType op_type, const uint64_t* op_properties);
189 
190  // Obtain the name of a state given its type.
191  static const std::string& GetStateName(StateType state_type);
192 };
