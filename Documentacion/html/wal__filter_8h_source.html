14 // WALFilter allows an application to inspect write-ahead-log (WAL)
15 // records or modify their processing on recovery.
16 // Please see the details below.
17 class WalFilter {
18  public:
19  enum class WalProcessingOption {
20  // Continue processing as usual
21  kContinueProcessing = 0,
22  // Ignore the current record but continue processing of log(s)
23  kIgnoreCurrentRecord = 1,
24  // Stop replay of logs and discard logs
25  // Logs won't be replayed on subsequent recovery
26  kStopReplay = 2,
27  // Corrupted record detected by filter
28  kCorruptedRecord = 3,
29  // Marker for enum count
30  kWalProcessingOptionMax = 4
31  };
32 
33  virtual ~WalFilter() {}
34 
35  // Provide ColumnFamily->LogNumber map to filter
36  // so that filter can determine whether a log number applies to a given
37  // column family (i.e. that log hasn't been flushed to SST already for the
38  // column family).
39  // We also pass in name->id map as only name is known during
40  // recovery (as handles are opened post-recovery).
41  // while write batch callbacks happen in terms of column family id.
42  //
43  // @params cf_lognumber_map column_family_id to lognumber map
44  // @params cf_name_id_map column_family_name to column_family_id map
45 
46  virtual void ColumnFamilyLogNumberMap(
47  const std::map<uint32_t, uint64_t>& cf_lognumber_map,
48  const std::map<std::string, uint32_t>& cf_name_id_map) {}
49 
50  // LogRecord is invoked for each log record encountered for all the logs
51  // during replay on logs on recovery. This method can be used to:
52  // * inspect the record (using the batch parameter)
53  // * ignoring current record
54  // (by returning WalProcessingOption::kIgnoreCurrentRecord)
55  // * reporting corrupted record
56  // (by returning WalProcessingOption::kCorruptedRecord)
57  // * stop log replay
58  // (by returning kStop replay) - please note that this implies
59  // discarding the logs from current record onwards.
60  //
61  // @params log_number log_number of the current log.
62  // Filter might use this to determine if the log
63  // record is applicable to a certain column family.
64  // @params log_file_name log file name - only for informational purposes
65  // @params batch batch encountered in the log during recovery
66  // @params new_batch new_batch to populate if filter wants to change
67  // the batch (for example to filter some records out,
68  // or alter some records).
69  // Please note that the new batch MUST NOT contain
70  // more records than original, else recovery would
71  // be failed.
72  // @params batch_changed Whether batch was changed by the filter.
73  // It must be set to true if new_batch was populated,
74  // else new_batch has no effect.
75  // @returns Processing option for the current record.
76  // Please see WalProcessingOption enum above for
77  // details.
78  virtual WalProcessingOption LogRecordFound(unsigned long long log_number,
79  const std::string& log_file_name,
80  const WriteBatch& batch,
81  WriteBatch* new_batch,
82  bool* batch_changed) {
83  // Default implementation falls back to older function for compatibility
84  return LogRecord(batch, new_batch, batch_changed);
85  }
86 
87  // Please see the comments for LogRecord above. This function is for
88  // compatibility only and contains a subset of parameters.
89  // New code should use the function above.
90  virtual WalProcessingOption LogRecord(const WriteBatch& batch,
91  WriteBatch* new_batch,
92  bool* batch_changed) const {
93  return WalProcessingOption::kContinueProcessing;
94  }
95 
96  // Returns a name that identifies this WAL filter.
97  // The name will be printed to LOG file on start up for diagnosis.
98  virtual const char* Name() const = 0;
99 };
