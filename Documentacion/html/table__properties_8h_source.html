14 // -- Table Properties
15 // Other than basic table properties, each table may also have the user
16 // collected properties.
17 // The value of the user-collected properties are encoded as raw bytes --
18 // users have to interprete these values by themselves.
19 // Note: To do prefix seek/scan in `UserCollectedProperties`, you can do
20 // something similar to:
21 //
22 // UserCollectedProperties props = ...;

56 // `TablePropertiesCollector` provides the mechanism for users to collect
57 // their own properties that they are interested in. This class is essentially
58 // a collection of callback functions that will be invoked during table
59 // building. It is construced with TablePropertiesCollectorFactory. The methods
60 // don't need to be thread-safe, as we will create exactly one
61 // TablePropertiesCollector object per table and then call it sequentially
62 class TablePropertiesCollector {
63  public:
64  virtual ~TablePropertiesCollector() {}
65 
66  // DEPRECATE User defined collector should implement AddUserKey(), though
67  // this old function still works for backward compatible reason.
68  // Add() will be called when a new key/value pair is inserted into the table.
69  // @params key the user key that is inserted into the table.
70  // @params value the value that is inserted into the table.
71  virtual Status Add(const Slice& /*key*/, const Slice& /*value*/) {
72  return Status::InvalidArgument(
73  "TablePropertiesCollector::Add() deprecated.");
74  }
75 
76  // AddUserKey() will be called when a new key/value pair is inserted into the
77  // table.
78  // @params key the user key that is inserted into the table.
79  // @params value the value that is inserted into the table.
80  virtual Status AddUserKey(const Slice& key, const Slice& value,
81  EntryType /*type*/, SequenceNumber /*seq*/,
82  uint64_t /*file_size*/) {
83  // For backwards-compatibility.
84  return Add(key, value);
85  }
86 
87  // Finish() will be called when a table has already been built and is ready
88  // for writing the properties block.
89  // @params properties User will add their collected statistics to
90  // `properties`.
91  virtual Status Finish(UserCollectedProperties* properties) = 0;
92 
93  // Return the human-readable properties, where the key is property name and
94  // the value is the human-readable form of value.
95  virtual UserCollectedProperties GetReadableProperties() const = 0;
96 
97  // The name of the properties collector can be used for debugging purpose.
98  virtual const char* Name() const = 0;
99 
100  // EXPERIMENTAL Return whether the output file should be further compacted
101  virtual bool NeedCompact() const { return false; }
102 };

122 // TableProperties contains a bunch of read-only properties of its associated
123 // table.
124 struct TableProperties {
125  public:
126  // the total size of all data blocks.
127  uint64_t data_size = 0;
128  // the size of index block.
129  uint64_t index_size = 0;
130  // the size of filter block.
131  uint64_t filter_size = 0;
132  // total raw key size
133  uint64_t raw_key_size = 0;
134  // total raw value size
135  uint64_t raw_value_size = 0;
136  // the number of blocks in this table
137  uint64_t num_data_blocks = 0;
138  // the number of entries in this table
139  uint64_t num_entries = 0;
140  // format version, reserved for backward compatibility
141  uint64_t format_version = 0;
142  // If 0, key is variable length. Otherwise number of bytes for each key.
143  uint64_t fixed_key_len = 0;
144  // ID of column family for this SST file, corresponding to the CF identified
145  // by column_family_name.
146  uint64_t column_family_id =
147  rocksdb::TablePropertiesCollectorFactory::Context::kUnknownColumnFamily;
148 
149  // Name of the column family with which this SST file is associated.
150  // If column family is unknown, `column_family_name` will be an empty string.
151  std::string column_family_name;
152 
153  // The name of the filter policy used in this table.
154  // If no filter policy is used, `filter_policy_name` will be an empty string.
155  std::string filter_policy_name;
156 
157  // user collected properties
158  UserCollectedProperties user_collected_properties;
159  UserCollectedProperties readable_properties;

166  // Aggregate the numerical member variables of the specified
