9 // An Env is an interface used by the rocksdb implementation to access
10 // operating system functionality like the filesystem etc. Callers
11 // may wish to provide a custom Env object when opening a database to
12 // get fine gain control; e.g., to rate limit file system operations.
13 //
14 // All Env implementations are safe for concurrent access from
15 // multiple threads without any external synchronization.

587  // Sync a file range with disk.

595  // PrepareWrite performs any necessary preparation for a write
596  // before the write actually occurs. This allows for pre-allocation
597  // of space on devices where it can result in less file
598  // fragmentation and/or less waste from over-zealous filesystem
599  // pre-allocation.
600  virtual void PrepareWrite(size_t offset, size_t len) {
601  if (preallocation_block_size_ == 0) {
602  return;
603  }
604  // If this write would cross one or more preallocation blocks,
605  // determine what the last preallocation block necesessary to
606  // cover this write would be and Allocate to that point.
607  const auto block_size = preallocation_block_size_;
608  size_t new_last_preallocated_block =
609  (offset + len + block_size - 1) / block_size;
610  if (new_last_preallocated_block > last_preallocated_block_) {
611  size_t num_spanned_blocks =
612  new_last_preallocated_block - last_preallocated_block_;
613  Allocate(block_size * last_preallocated_block_,
614  block_size * num_spanned_blocks);
615  last_preallocated_block_ = new_last_preallocated_block;
616  }
617  }
618 
619  protected:
620  /*
621  * Pre-allocate space for a file.
622  */
623  virtual Status Allocate(uint64_t offset, uint64_t len) {
624  return Status::OK();
625  }
626 
627  size_t preallocation_block_size() { return preallocation_block_size_; }
628 
629  private:
630  size_t last_preallocated_block_;
631  size_t preallocation_block_size_;
632  // No copying allowed
633  WritableFile(const WritableFile&);
634  void operator=(const WritableFile&);
635 
636  protected:
637  friend class WritableFileWrapper;
638  friend class WritableFileMirror;
639 
640  Env::IOPriority io_priority_;
641 };
